// GET /users/:id RequestObject:

[{
  name: 'Professor',
  id: 'xxxxx-xxxx-xxxxxxx',
  currentLocation: 'Starbucks',
  favoriteLocation: 'Tim Hortons',
  username: 'Professor1234',
  onlineStatus: 'Yes',
}]
